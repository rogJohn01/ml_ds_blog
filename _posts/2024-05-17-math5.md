$$
f'(x) = \lim_{\substack{h \to 0}} \frac{f(x + h) - f(x)}{h}
$$

$$
f'(x) = \lim_{\substack{h \to 0}} \frac{e^{x + h} - e^x}{h}
$$

$$
f'(x) = \lim_{\substack{h \to 0}} \frac{e^x \cdot e^h - e^x}{h}
$$

$$
f'(x) = \lim_{\substack{h \to 0}} e^x \cdot \frac{e^h - 1}{h}
$$
